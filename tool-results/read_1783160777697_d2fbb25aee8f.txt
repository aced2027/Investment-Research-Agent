    40→function generateSparkline(price: number, changePercent: number): number[] {
    41→  const points: number[] = []
    42→  let p = price * (1 - changePercent / 100 * 3)
    43→  const trend = changePercent > 0 ? 0.3 : -0.3
    44→  for (let i = 0; i < 7; i++) {
    45→    p += (trend + (i / 7) * (changePercent > 0 ? 0.5 : -0.5)) * (price / 100)
    46→    points.push(Math.round(p * 100) / 100)
    47→  }
    48→  points[6] = price
    49→  return points
    50→}

   246→export function DashboardScreen() {
   247→  const [indices, setIndices] = useState<MarketIndex[]>([])
   248→  const [gainers, setGainers] = useState<TopMover[]>([])

   250→  const [insights, setInsights] = useState<AIInsight[]>([])
   251→  const [news, setNews] = useState<NewsItem[]>([])
   252→  const [loading, setLoading] = useState(true)
   253→  const [error, setError] = useState<string | null>(null)
   254→
   255→  useEffect(() => {
   256→    let cancelled = false
   257→
   258→    async function load() {
   259→      try {
   260→        setLoading(true)
   261→        setError(null)
   262→        const data = await getDashboardSnapshot()
   263→        if (cancelled) return
   264→        setIndices(data.indices)
   265→        setGainers(data.movers.gainers)
   266→        setLosers(data.movers.losers)
   267→        setNews(data.news)
   268→        // Generate insights from top gainer data
   269→        if (data.movers.gainers.length > 0) {
   270→          const topGainer = data.movers.gainers[0]
   271→          try {
   272→            const quote = await getStockQuote(topGainer.symbol)
   273→            if (cancelled) return
   274→            const recs = await getRecommendations(topGainer.symbol).catch(() => [])
   275→            if (cancelled) return
   276→            const insider = await getInsiderSentiment(topGainer.symbol).catch(() => [])
   277→            if (cancelled) return
   278→            const generated = generateInsightsFromData(quote, recs, insider)
   279→            setInsights(generated.length > 0 ? generated : [
   280→              { type: 'opportunity' as const, text: `${topGainer.symbol} leading gainers today at +${topGainer.change.toFixed(2)}%. Strong momentum across technology sector.`, confidence: 75 }

   288→        }
   289→      } catch (err) {
   290→        if (cancelled) return

   293→        if (!cancelled) {
   294→          setLoading(false)
   295→        }
   296→      }
   297→    }
   298→    load()
   299→    // Refresh every 30 seconds

   302→      cancelled = true

   305→  }, [])

   318→      symbol: g.symbol,
   319→      price: g.price,
   320→      change: g.change,

   328→  })), [])

   332→    return (
   333→      <div className="space-y-5">
   334→        <Skeleton className="h-8 w-64" />
   335→        <div className="grid grid-cols-2 lg:grid-cols-4 gap-3">
