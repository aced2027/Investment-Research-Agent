    40тЖТ    40тЖТfunction generateSparkline(price: number, changePercent: number): number[] {
    41тЖТ    41тЖТ  const points: number[] = []
    42тЖТ    42тЖТ  let p = price * (1 - changePercent / 100 * 3)
    43тЖТ    43тЖТ  const trend = changePercent > 0 ? 0.3 : -0.3
    44тЖТ    44тЖТ  for (let i = 0; i < 7; i++) {
    45тЖТ    45тЖТ    p += (trend + (i / 7) * (changePercent > 0 ? 0.5 : -0.5)) * (price / 100)
    46тЖТ    46тЖТ    points.push(Math.round(p * 100) / 100)
    47тЖТ    47тЖТ  }
    48тЖТ    48тЖТ  points[6] = price
    49тЖТ    49тЖТ  return points
